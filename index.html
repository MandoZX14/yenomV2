/******************************************************** * ======================= STATE ======================== ********************************************************/ const STATE = { pairs: ["BTC","ETH","SOL","BNB","XRP","DOGE"], prices: {}, currentPair: localStorage.getItem('pair') || 'BTC', capital: Number(localStorage.getItem('cap') ?? 20), goal: Number(localStorage.getItem('goal') ?? 100), ws: null }; /******************************************************** * ======================= UTILS ======================== ********************************************************/ const $ = id => document.getElementById(id); const formatPrice = n => { if(!Number.isFinite(n)) return '—'; return n >= 1000 ? n.toLocaleString('fr-FR',{maximumFractionDigits:2}) : n.toLocaleString('fr-FR',{maximumFractionDigits:6}); }; const toast = (title, text) => { $("toastTitle").textContent = title; $("toastText").textContent = text; $("toastBox").classList.add('show'); setTimeout(()=> $("toastBox").classList.remove('show'), 1500); }; /******************************************************** * ======================= WEBSOCKET ==================== ********************************************************/ function connectWS(){ if(STATE.ws) STATE.ws.close(); const streams = STATE.pairs.map(p => p.toLowerCase()+'usdt@ticker').join('/'); const url = `wss://stream.binance.com:9443/stream?streams=${streams}`; STATE.ws = new WebSocket(url); STATE.ws.onopen = () => { $("ws-st").textContent = "CONNECTÉ"; $("ws-st").style.color = "var(--green)"; }; STATE.ws.onclose = () => { $("ws-st").textContent = "RECONNECT..."; setTimeout(connectWS, 2000); }; STATE.ws.onerror = () => { STATE.ws.close(); }; STATE.ws.onmessage = e => { const data = JSON.parse(e.data).data; if(!data?.s) return; const sym = data.s.replace('USDT',''); STATE.prices[sym] = { price: +data.c, change: +data.P, high: +data.h, low: +data.l, vol: +data.q }; updateTicker(sym); if(sym === STATE.currentPair){ updateMain(); } }; } /******************************************************** * ======================= UI =========================== ********************************************************/ function buildTicker(){ $("tickerBar").innerHTML = STATE.pairs.map(p => `
` ).join(''); } function updateTicker(sym){ const el = $("ti-"+sym); const d = STATE.prices[sym]; if(!el || !d) return; el.innerHTML = ` ${sym} ${formatPrice(d.price)} `; } function updateMain(){ const d = STATE.prices[STATE.currentPair]; if(!d) return; $("mainPairName").textContent = STATE.currentPair; $("main-price").textContent = "$"+formatPrice(d.price); const badge = $("chg-badge"); badge.textContent = (d.change>=0?'+':'')+d.change.toFixed(2)+'%'; badge.className = "badge " + (d.change>=0?'green':'red'); $("hi-v").textContent = "$"+formatPrice(d.high); $("lo-v").textContent = "$"+formatPrice(d.low); $("vo-v").textContent = (d.vol/1000).toFixed(1)+'K'; } /******************************************************** * ======================= WALLET ======================= ********************************************************/ function updateWallet(){ $("cap-v").textContent = STATE.capital.toFixed(2); $("goal-v").textContent = STATE.goal.toFixed(2); const pct = Math.min((STATE.capital/STATE.goal)*100,100); $("goal-bar").style.width = pct+"%"; $("goal-pct").textContent = pct.toFixed(1)+"%"; } function saveWallet(){ const val = Number($("walletInput").value); if(!val || val <= 0) return toast("ERREUR","Capital invalide"); STATE.capital = val; localStorage.setItem('cap', val); updateWallet(); toast("OK","Capital mis à jour"); } /******************************************************** * ======================= LOGIC ======================== ********************************************************/ function switchPair(p){ STATE.currentPair = p; localStorage.setItem('pair', p); document.querySelectorAll('.titem').forEach(el => el.classList.remove('active') ); $("ti-"+p).classList.add('active'); updateMain(); } /******************************************************** * ======================= OPPS ========================= ********************************************************/ function buildOpps(){ const ops = [ {asset:'SOL',dir:'LONG',conf:91}, {asset:'BTC',dir:'SHORT',conf:84} ]; $("oppsGrid").innerHTML = ops.map(o=>`
${o.asset} ${o.dir}
${o.conf}%
`).join(''); $("oppsCount").textContent = ops.length; } /******************************************************** * ======================= TRADE ======================== ********************************************************/ function openTrade(asset, dir, conf){ $("tradeAsset").textContent = `${asset} • ${dir}`; $("tradeOverlay").classList.add("show"); } function closeTrade(){ $("tradeOverlay").classList.remove("show"); } /******************************************************** * ======================= INIT ========================= ********************************************************/ function init(){ buildTicker(); buildOpps(); updateWallet(); $("walletInput").value = STATE.capital; connectWS(); setTimeout(updateMain, 500); } document.addEventListener("DOMContentLoaded", init);
